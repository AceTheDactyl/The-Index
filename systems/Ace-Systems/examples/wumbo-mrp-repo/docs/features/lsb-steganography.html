Encoding Process

// 1. Create pattern from state
const pattern = WumboMRP.createPattern({ z: 0.866, regionId: 47 });

// 2. Get image data from canvas
let imageData = ctx.getImageData(0, 0, width, height);

// 3. Encode pattern into image
imageData = WumboMRP.encode(imageData, pattern, { redundancy: 16 });

// 4. Write back to canvas
ctx.putImageData(imageData, 0, 0);
