Live Application Example

// Update at 30fps
function updateLoop() {
    const z = getSensorReading();  // ~1ms
    const physics = WumboMRP.computePhysics(z);  // < 0.5ms
    updateUI(physics);

    // Encode snapshot every 500ms
    if (frameCount % 15 === 0) {
        encodeSnapshot(z);  // ~2ms async
    }

    requestAnimationFrame(updateLoop);
}
